\begin{frame}{Goal}
    \begin{thm}
        Let $G = \gen{X} \leq \sym \Omega$ be a primitive group
        of {\color{blue} PA type}.
        The normaliser
        $N_{\sym \Omega}(G)$
        can be computed in {\color{blue} quasipolynomial} time
        $O(n ^ 3 \cdot 2 ^ {2 \log n \log \log n} \cdot \abs X)$.
    \end{thm}

\end{frame}
